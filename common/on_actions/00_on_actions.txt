# GOL巴尔干战争On_Action系统
# 基于即时触发机制的和平协议系统

on_action = {
	# 投降立即拦截（设置flag）
	on_capitulation_immediate = {
		effect = {
			# 瓦拉几亚投降
			if = {
				limit = {
					FROM = { tag = ROM }
					ROOT = { tag = HUN }
					NOT = { has_global_flag = GOL_flag_hungarian_victory }
					NOT = { has_global_flag = GOL_flag_balkan_alliance_victory }
				}
				set_global_flag = GOL_flag_romania_defeated
			}
			
			# 塞尔维亚投降
			if = {
				limit = {
					FROM = { tag = YUG }
					ROOT = { tag = HUN }
					NOT = { has_global_flag = GOL_flag_hungarian_victory }
					NOT = { has_global_flag = GOL_flag_balkan_alliance_victory }
				}
				set_global_flag = GOL_flag_yugoslavia_defeated
			}
			
			# 匈牙利投降
			if = {
				limit = {
					FROM = { tag = HUN }
					ROOT = { OR = { tag = ROM tag = YUG } }
					NOT = { has_global_flag = GOL_flag_hungarian_victory }
					NOT = { has_global_flag = GOL_flag_balkan_alliance_victory }
				}
				set_global_flag = GOL_flag_hungary_defeated
			}
		}
	}
	
	# 和平会议开始前拦截（执行和平脚本）
	on_before_peace_conference_start = {
		effect = {
			# 匈牙利阵营胜利
			if = {
				limit = {
					ROOT = { tag = HUN }
					FROM = { OR = { tag = ROM tag = YUG } }
					NOT = { has_global_flag = GOL_flag_hungarian_victory }
					NOT = { has_global_flag = GOL_flag_balkan_alliance_victory }
					# 检查两个国家是否都被击败
					OR = {
						AND = {
							FROM = { tag = ROM }
							OR = {
								has_global_flag = GOL_flag_yugoslavia_defeated
								YUG = {
									all_core_state = {
										OR = {
											is_controlled_by = HUN
											is_controlled_by = MAC
										}
									}
								}
							}
						}
						AND = {
							FROM = { tag = YUG }
							OR = {
								has_global_flag = GOL_flag_romania_defeated
								ROM = {
									all_core_state = {
										OR = {
											is_controlled_by = HUN
											is_controlled_by = MAC
										}
									}
								}
							}
						}
					}
				}
				GOL_hungarian_victory_peace = yes
				news_event = { id = GOL_balkan.1 }
			}
			
			# 巴尔干同盟胜利
			if = {
				limit = {
					FROM = { tag = HUN }
					ROOT = { OR = { tag = ROM tag = YUG } }
					NOT = { has_global_flag = GOL_flag_hungarian_victory }
					NOT = { has_global_flag = GOL_flag_balkan_alliance_victory }
				}
				GOL_balkan_alliance_victory_peace = yes
				news_event = { id = GOL_balkan.2 }
			}
		}
	}
}
